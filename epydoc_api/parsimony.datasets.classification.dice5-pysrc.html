19      """Generate classification samples (images + target variable) and beta. 
20      Call make_regression_struct then apply the logistic function: 
21      proba = 1. / (1 + exp(-(X * beta + noise)), then 
22      y = 1 if proba >= 0.5 else 0 
23   
24      Parameters 
25      ---------- 
26      See make_regression_struct, exept for r2 which is replaced by snr. 
27   
28      snr: Float. Default 2. Signal to noise ratio: std(X * beta) / std(noise) 
29              in 1. / (1 + exp(-(X * beta + noise)) 
30   
31      sigma_logit: Float. Default is 5. The standard deviation of 
32          logit = X * beta + noise. Large sigma_logit promotes sharp shape of 
33          logistic function, ie: probabilities are close to 0 or 1, which 
34          increases likekihood. 
35   
36      Returns 
37      ------- 
38      X3d: Numpy array of shape [n_sample, shape]. The input features. 
39   
40      y: Numpy array of shape [n_sample, 1]. The target variable. 
41   
42      beta3d: Float array of shape [shape]. It is the beta such that 
43      y = 1. / (1 + exp(-(X * beta + noise)). 
44   
45      proba: Numpy array of shape [n_sample, 1]. Samples posterior probabilities. 
46   
47      See also 
48      -------- 
49      make_regression_struct 
50   
51      Examples 
52      -------- 
53      >>> import numpy as np 
54      >>> import matplotlib.pyplot as plot 
55      >>> from  parsimony import datasets 
56      >>> n_samples = 100 
57      >>> shape = (11, 11, 1) 
58      >>> X3d, y, beta3d, proba = datasets.classification.dice5.load(n_samples=n_samples, 
59      ...     shape=shape, snr=5, random_seed=1) 
60      >>> print "Likelihood=", np.prod(proba[y.ravel()==1]) * np.prod(1-proba[y.ravel()==0]) 
61      Likelihood= 3.85343753829e-06 
62      >>> X3d, y, beta3d, proba = datasets.classification.dice5.load(n_samples=n_samples, 
63      ...     shape=shape, sigma_logit=5., random_seed=1) 
64      >>> print "Likelihood=", np.prod(proba[y.ravel()==1]) * np.prod(1-proba[y.ravel()==0]) 
65      Likelihood= 2.19102268035e-06 
66      """ 
67      X3d, y, beta3d = dice5regression.load(n_samples=n_samples, shape=shape, r2=1., 
68                                              random_seed=random_seed, **kwargs) 
69      X = X3d.reshape((n_samples, np.prod(shape))) 
70      beta = beta3d.ravel() 
71      coef = float(sigma_logit) / np.sqrt(snr ** 2 + 1) 
72      beta *= coef * snr / np.std(np.dot(X, beta)) 
73      if random_seed is not None:  # If random seed, save current random state 
74          rnd_state = np.random.get_state() 
75          np.random.seed(random_seed) 
76      noise = coef * np.random.normal(0, 1, X.shape[0]) 
77      if random_seed is not None:   # If random seed, restore random state 
78          np.random.set_state(rnd_state) 
79      logit = np.dot(X, beta) + noise 
80      #np.std(np.dot(X, beta)) / np.std(noise) 
81      #np.std(logit) 
82      proba = 1. / (1. + np.exp(-logit)) 
83      y = np.ones(y.shape) 
84      y[proba < .5] = 0 
85      return X3d, y, beta3d, proba
